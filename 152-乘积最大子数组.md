func max(a, b int) int {
    if a > b {
        return a
    } else {
        return b
    }
}
func min(a, b int) int {
    if a < b {
        return a
    } else {
        return b
    }
}
func maxProduct(nums []int) int {
    n := len(nums)
    maxF := nums[0]
    minF := nums[0]
    ans := nums[0]
    for i := 1; i < n; i++ {
        tempMaxF := max(nums[i], max(maxF * nums[i], minF * nums[i]))
        minF = min(nums[i], min(maxF * nums[i], minF * nums[i]))
        maxF = tempMaxF
        if maxF > ans {
            ans = maxF
        }
    }
    return ans
}
